# 題目

給一個字串,找出最大不重複的字元長度

例如假設字串為 "abcabcbb",那最大不重複的字元為 "abc", 所以長度是 3

例如假設字串為 "bbbbb",那最大不重複的字元為 "b", 所以長度是 1

# 想法一(TLE)
```Swift
/*
直觀想法,先取出字串第一個字元,並逐一往後比較,算出最大不重複字元長度
接著取出字串第二個字元,一樣逐一往後比對,不過此方法最後會超過時間限制
*/
class Solution {
    func lengthOfLongestSubstring(s: String) -> Int {
        let inputStrLen = s.characters.count
        var maxLen      = 1
        for idx in 0..<inputStrLen {
            let strIdx      = s.startIndex.advancedBy(idx)
            var maps        = [Character: Int]()
            var strLen      = 1
            maps[s.characters[strIdx]]   = 1
            //print("idx(\(idx)) = map:\(maps)")
            for i in (idx+1)..<inputStrLen {
                let cmpIdx  = s.startIndex.advancedBy(i)
                if let _ = maps[s.characters[cmpIdx]] {
                    if maxLen <= strLen {
                        maxLen = strLen
                    }
                    //print("@maps:\(maps) strLen:\(strLen)")
                    break
                } else {
                    maps[s.characters[cmpIdx]] = 1
                    strLen += 1
                    //print("!maps:\(maps) strLen:\(strLen)")
                }
            }
        }
        return maxLen
    }
}
let solution    = Solution()
let stringTest  = "aasdffewbw"
print(solution.lengthOfLongestSubstring(stringTest))
```

# 想法二(TLE)
```Swift
/*
原本想說是雙迴圈的關係,所以改良為使用單一迴圈,不過一樣會導致 TLE
*/
class Solution {
    func lengthOfLongestSubstring(s: String) -> Int {
        let inputStrLen = s.characters.count
        if inputStrLen == 0 {
            return 0
        }
        var maps        = [Character: Int]() 
        var maxLen      = 1  /* 最大連續長度 */
        var start       = 0  /* 指向 start 位置 */
        var end         = 1  /* 指向 end   位置,會一直往右移動 */       
        
        /* 字串第一個字 */
        let strIdx  = s.startIndex.advancedBy(0)
        let nowChar = s.characters[strIdx]
        maps[nowChar] = 0
    
        while end < inputStrLen {
            let cmpIdx  = s.startIndex.advancedBy(end)
            let cmpChar = s.characters[cmpIdx]
        
            if let val = maps[cmpChar] {
                if val >= start {
                    start   = val + 1;
                }
            }
            maxLen          = max(maxLen, end - start + 1)
            maps[cmpChar]   = end
            end             += 1
        }
        return maxLen
    }
}
```

    }

# 想法四(52-60 ms)
```Swift
/*
後來發現是 Swift 本身語言的問題,導致太慢的原因出在 s.startIndex.advancedBy(end)
此方法的作用在取得字串中指定索引的位置,例如 s = "abcabcbb" 則 s.startIndex.advancedBy(3) = 'a'(字元類型)
不過一但字串非常大就會很慢

解決的方法是,一開始就先把輸入字串轉為另一個字串陣列 let strArray = Array(s.unicodeScalars)
例如 s = "abcabcbb" 則 strArray = ["a","b","c","a","b","c","b","b"]

想法四主要是使用 Int Array,此陣列的索引數及對應字元的 unicode (例如 a = 97)
而陣列的數值及為字元重複的次數,例如碰到 a 兩次則 strArray[97] = 2
*/
extension Character
{
    func unicode() -> Int
    {
        let characterString = String(self)
        let scalars = characterString.unicodeScalars
        return Int(scalars[scalars.startIndex].value)
    }
}
class Solution {
    func lengthOfLongestSubstring(s: String) -> Int {
        guard s.characters.count > 0 else {
            // 當沒有傳入字串時直接回傳 0
            return 0
        }
        var res         = 0
        var left        = 0
        var prev        = [Int](count: 300, repeatedValue: -1)
        let strArray    = Array(s.unicodeScalars)

        for var i in 0..<s.characters.count {
            var nowChar = Character(strArray[i])
            if prev[nowChar.unicode()] >= left {
                left = prev[nowChar.unicode()] + 1
            }
            prev[nowChar.unicode()] = i
            res = (res < i - left + 1)  ? (i - left + 1) : res
        }
        return res
    }
}
```

# 想法五(76 ms)
```Swift
/*
根據想法四改變而來,目的是希望取消 extension Character 的 unicode() 方法
直接以 map 來判斷儲存字元重複的次數(如果沒遇到的字元 map 的 key 一定找不到)
但此法比想法四還慢,也許是 map 型別的處理比 Array 慢
*/
class Solution {
    func lengthOfLongestSubstring(s: String) -> Int {
        guard s.characters.count > 0 else {
            // 當沒有傳入字串時直接回傳 0
            return 0
        }
        var maps        = [Character: Int]() 
        var maxLen      = 1  /* 最大連續長度 */
        var start       = 0  /* 指向 start 位置 */
        var end         = 1  /* 指向 end   位置,會一直往右移動 */       
        let strArray    = Array(s.unicodeScalars)
        
        maps[Character(strArray[0])] = 0
        for var end in 0..<s.characters.count {
            if let val = maps[Character(strArray[end])] {
                start = (val >= start) ? (val + 1) : start
            }
            maxLen                          = max(maxLen, end - start + 1)
            maps[Character(strArray[end])]  = end
        }
        return maxLen
    }
}
```
